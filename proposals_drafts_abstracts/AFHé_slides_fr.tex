\begin{frame}{Motivations}

\begin{itemize}

  \item Juin 2025: 117,3 millions de personnes forcées de fuir leur domicile dans le monde entier (HCR, 2025). 
\vspace{0.4cm}
  \item La persécution à motivation politique génère une sélection idéologique dans l'exil.
\vspace{0.4cm}
  \item Les sociétés d'accueil connaissent une hausse des attitudes anti-immigrants (OIM, 2024)
\vspace{0.4cm}
  \item Étudier comment les réactions politiques des sociétés d'accueil émergent, se propagent et persistent est essentiel pour comprendre les conséquences politiques à long terme de ces migrations forcées.
\end{itemize}


\end{frame}

\begin{frame}{Dans ce projet}
\vspace{0.1in}
\begin{itemize}
  \item \textbf{Déplacement de masse de la population espagnole et des forces républicaines} en 1939 vers la France afin d'estimer les effets politiques de l'accueil des réfugiés politiques à différents horizons temporels.
  \vspace{0.15in}
  \item Comment les réfugiés politiques ont ils influencés les croyances politiques et l'action collective dans les communautés d'accueil ?
  \vspace{0.2in}
  \item Effets à \textbf{court terme :} 
  \vspace{0.05in}
  \begin{itemize}
      \item Les communes exposées aux camps de réfugiés montrent systématiquement un \textbf{moins grand soutien pour les partis de gauche};
      \vspace{0.1in}
      \item Influence positive sur l'implication dans les mouvements de Résistance pendant la Seconde Guerre mondiale. 
      \vspace{0.1in}
      \item Des preuves suggestives indiquent que le type de contact est important.
  \end{itemize}. 
\end{itemize}

\end{frame}

\section{Contexte}

\begin{frame}
  \frametitle{La Guerre civile espagnole (1936-1939)}
  \footnotesize
  \begin{itemize}
    \item \textbf{Contexte :} Tensions entre les républicains et les nationalistes, jeune République, nombreuses instabilités.
    \vspace{0.1in}
    \item 17 juillet 1936 : Coup d'État militaire mené par Franco, divisant le pays.
    \vspace{0.1in}
    \item \textbf{Deux Camps :} 
      \begin{itemize}
        \item Républicains : Coalition de gauche (communistes, socialistes, anarchistes, syndicats, etc.).
        \item Nationalistes : Menés par Franco (militaires, monarchistes, fascistes).
      \end{itemize}
      \vspace{0.1in}
    \item \textbf{Avril 1939} : Victoire nationaliste, dictature de Franco jusqu'à la transition vers la démocratie en 1975.
  \end{itemize}
\end{frame}

  \begin{itemize}
  \item Aucune installation permanente à l'avance $\Longrightarrow$ sélection des sites utilisé pour l'installation des camps de réfugiés précipitée \citep{DreyfusArmand1999}
  \item Les réfugiés espagnols sont forcés de vivre là où ils ont été assignés, sous la surveillance des autorités. (\cite{perez_rodriguez_indeseables_2022})

\begin{frame}{Politique de dispersion - Contexte en France}
Deux ``types" d'accueil: 
\vspace{0.1in}
\begin{itemize}
  \item Les \textit{combattants} étaient principalement localisés dans les camps près de la frontière.
  \item La \textit{population civile} a été transférée dans des centres d'accueil dispersés à travers la France. 
  \vspace{0.2in}
  \item Les autorités françaises encouragent le retour en Espagne (Accords Bérard-Jordana, février 1939) ou l'émigration vers des pays tiers.
  % le CIARE a calculé qu'environ 400 000 personnes avaient traversé la frontière vers la France. Parmi elles, 60 000 étaient déjà retournées dans la péninsule en juillet 1939. À la fin de 1939, le nombre d'exilés en France était de 140 000 (plus de sélection politique). Certains pays d'Amérique latine ont accueilli les réfugiés. 
  \vspace{0.2in}
  \item Beaucoup de ceux qui sont restés ont été forcés d'entrer dans des brigades de travail ou la Légion étrangère. Avec le déclenchement de la guerre, 15 000 se sont joints à la Résistance. 
  %Fin 1939, environ 50 000 à 60 000 Espagnols servaient dans ces entreprises de travail militarisées à travers la France et 7000 rejoignent la Légion étrangère \citep{academicoup} % EXPLIQUER ICI L'OCCUPATION. Des milliers de républicains espagnols ont ensuite été déportés dans les camps de concentration nazis—le plus tristement célèbre étant Mauthausen, où environ 7 000 sont morts.
\end{itemize}




\end{frame}

\begin{frame}{Stratégie empirique}
\textit{Cross section} (résultats pendant la guerre) : 
\begin{equation}\label{equation:refugees_resistance}
    y_{m} = \beta \cdot \text{Refugees}_{m} + X_{m} + \delta_{m} + \varepsilon_{m}
\end{equation}
\\
\vspace{0.2in}
\\
\textit{Difference-in-differences} (participation politique avant et après la guerre) ;
\begin{equation}
     y_{mt} = \beta (host \ refugees_{m} \times Post_t)+ \alpha_{m} + \delta_{t,d} + \epsilon_{mt}
\end{equation}

\begin{itemize}
    \item  $y_{mt}$  résultat politique dans la municipalité $m$ au temps $t$
    \item $host refugees_{m}$ égale 1 si la municipalité a accueilli des réfugiés
    \item $Post_t$ égale 1 pour les périodes après 1936
    \item $\alpha_{m}$ et $\delta_{t,d}$ sont les effets fixes de la municipalité et du temps.
\end{itemize}


\end{frame}
