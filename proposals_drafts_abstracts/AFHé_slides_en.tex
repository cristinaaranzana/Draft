\begin{frame}{Motivations}

\begin{itemize}

  \item As of the end of June 2025, 117.3 million people had been forced to flee their homes globally (UNHCR, 2025). 
\vspace{0.4cm}
  \item Politically motivated persecution generates ideological selection into exile.
\vspace{0.4cm}
  \item At the same time, receiving societies exhibit a sustained rise in anti-immigrant attitudes (IOM, 2024)
\vspace{0.4cm}
  \item Studying how host political backlash emerges, propagates, and persists matters for understanding the long-run political consequences of forced migration.
\end{itemize}

\begin{frame}{In this project}

\begin{itemize}
  \item We study the \textbf{1939 mass displacement of Spanish Population and Republicans forces} into France in order to estimate the political effects of welcoming political refugees at different time horizons.
  \vspace{0.15in}
  \item \textbf{Research Question:} How do politically selected refugees shape political believes and collective action in host communities?
  \vspace{0.2in}
  \item \textbf{(For now) results on short term outcomes:} 
  \vspace{0.05in}
  \begin{itemize}
      \item Communes exposed to refugee camps display systematically \textbf{lower support for left-wing parties};
      \vspace{0.1in}
      \item Positive influence on the involvement into the R\textbf{ésistance movements} during WWII. 
      \vspace{0.1in}
      \item Suggestive evidence indicates that the type of contact matters.
  \end{itemize}. 
\end{itemize}


\end{frame}

\begin{frame}{Empirical Strategy}
Cross section (outcomes during the war): 
\begin{equation}\label{equation:refugees_resistance}
    y_{m} = \beta \cdot \text{Refugees}_{m} + X_{m} + \delta_{m} + \varepsilon_{m}
\end{equation}
\\
\vspace{0.2in}
\\
Diff-in-differences (political participation before and after the War);
\begin{equation}
     y_{mt} = \beta (host \ refugees_{m} \times Post_t)+ \alpha_{m} + \delta_{t,d} + \epsilon_{mt}
\end{equation}

\begin{itemize}
    \item  $y_{mt}$  political outcome in municipality $m$ at time $t$
    \item $host refugees_{m}$ equals 1 if the municipality hosted refugees
    \item $Post_t$ equals 1 for periods after 1936
    \item $\alpha_{m}$ and $\delta_{t,d}$ are municipality and time fixed effect.
\end{itemize}


\end{frame}
